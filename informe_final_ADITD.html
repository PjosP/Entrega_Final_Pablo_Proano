# FunciÃ³n para eliminar atÃ­picos usando IQR
eliminar_outliers_IQR <- function(x) {
  Q1 <- quantile(x, 0.25, na.rm = TRUE)
  Q3 <- quantile(x, 0.75, na.rm = TRUE)
  IQR_val <- Q3 - Q1
  limite_inf <- Q1 - 1.5 * IQR_val
  limite_sup <- Q3 + 1.5 * IQR_val
  x >= limite_inf & x <= limite_sup
}

filtro_IVA  <- eliminar_outliers_IQR(base_seleccionada$IVA)
filtro_ADV  <- eliminar_outliers_IQR(base_seleccionada$ADVALOREM)
filtro_ICE  <- eliminar_outliers_IQR(base_seleccionada$ICE_ADVALOREM)
filtro_FOD  <- eliminar_outliers_IQR(base_seleccionada$FODINFA)

filtro_total <- filtro_IVA & filtro_ADV & filtro_ICE & filtro_FOD

base_seleccionada_sin_outliers <- base_seleccionada[filtro_total, ]
